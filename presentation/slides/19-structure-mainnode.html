class ShoppinkkiMainNode(Node):
    def __init__(self):
        super().__init__('shoppinkki_main')

        # ① 상태머신 초기화
        self.sm = ShoppinkkiFSM(...)

        # ② 하드웨어 초기화
        self.hw = HWController(...)

        # ③ 인식 모듈 초기화
        self.doll_detector = DollDetector(...)

        # ④ BT 생성 후 BTRunner에 등록
        self.bt_runner = BTRunner(sm=self.sm, bts={...})

        # ⑤ ROS2 타이머 (BT tick 10Hz, 상태 발행 1Hz)
        self.create_timer(0.1, self._bt_tick_callback)
        self.create_timer(1.0, self._status_pub_callback)

        # ⑥ 별도 스레드 (AI 추론, 카메라 스트림)
        self._ai_thread = Thread(target=self._ai_loop, daemon=True)
        self._cam_thread = Thread(target=self._cam_loop, daemon=True)
        self._ai_thread.start(); self._cam_thread.start()

# ⑦ ROS2 루프 진입
def main():
    rclpy.init()
    rclpy.spin(ShoppinkkiMainNode())
1

ShoppinkkiFSM — 상태 전이 시 콜백으로 HW·BT를 조율합니다.

2

HWController — LCD·LED·버저를 추상화해 상태별 출력을 담당합니다.

3

DollDetector — AI 서버에 연결해 인형 감지·ReID 매칭 결과를 제공합니다.

4

BTRunner — SM 상태가 바뀌면 해당 BT를 선택·실행합니다.

5

create_timer — BT를 10Hz로 tick, 상태를 1Hz로 발행합니다.

6

Thread — AI 추론·카메라 스트림을 별도 스레드에서 병렬 실행합니다.

7

rclpy.spin() — ROS2 이벤트 루프. 토픽·타이머·액션을 처리합니다.
